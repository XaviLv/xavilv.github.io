def sigmoid(z):
    a = 1 / (1 + np.exp(-z))
    return a

def plot_sigmoid():
    z = np.arange(-8, 8, 0.2)
    a = sigmoid(z)
    plt.plot(z, a)
    plt.title('Sigmoid Function')
    plt.show()

def plot_tanh():
    z = np.arange(-8, 8, 0.2)
    a = (np.exp(z) - np.exp(-z)) / (np.exp(z) + np.exp(-z))
    plt.plot(z, a)
    plt.title('Tanh Function')
    plt.show()

def plot_ReLU():
    z = np.arange(-8, 8, 0.2)
    a = list(map(lambda x: x if x > 0 else 0, z))
    plt.plot(z, a)
    plt.title('ReLU Function')
    plt.show()
